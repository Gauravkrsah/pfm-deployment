EXPENSE_PATTERNS = [

\subsection{Currency Preprocessing}

\begin{lstlisting}[language=Python, caption=Currency Format Preprocessing]
def _preprocess_text(self, text: str) -> str:
    """Convert lakh, crore, k to numeric values."""
    
    # Handle "1 lakh" -> "100000"
    text = re.sub(
        r'(\d+(?:\.\d+)?)\s*lakh',
        lambda m: str(int(float(m.group(1)) * 100000)),
        text, flags=re.IGNORECASE
    )
    
    # Handle "1 crore" -> "10000000"
    text = re.sub(
        r'(\d+(?:\.\d+)?)\s*crore',
        lambda m: str(int(float(m.group(1)) * 10000000)),
        text, flags=re.IGNORECASE
    )
    
    # Handle "5k" -> "5000"
    text = re.sub(
        r'(\d+(?:\.\d+)?)\s*k\b',
        lambda m: str(int(float(m.group(1)) * 1000)),
        text, flags=re.IGNORECASE
    )
    
    return text
\end{lstlisting}

\section{RAG Query Processing}

\begin{lstlisting}[language=Python, caption=RAG Context Preparation]
def _prepare_expense_context(self, expenses: List[dict], 
                             query: str) -> str:
    """Prepare expense context for LLM."""
    
    # Calculate summary statistics
    total = sum(e['amount'] for e in expenses)
    count = len(expenses)
    
    # Group by category
    by_category = {}
    for e in expenses:
        cat = e.get('category', 'Other')
        if cat not in by_category:
            by_category[cat] = []
        by_category[cat].append(e)
    
    # Build context string
    context = f"""
    EXPENSE SUMMARY:
    - Total expenses: {count} transactions
    - Total amount: Rs. {total:,.2f}
    
    BY CATEGORY:
    """
    
    for cat, items in sorted(by_category.items(), 
                             key=lambda x: -sum(i['amount'] 
                             for i in x[1])):
        cat_total = sum(i['amount'] for i in items)
        context += f"\n    {cat}: Rs. {cat_total:,.2f} ({len(items)} items)"
    
    # Add recent expenses
    context += "\n\n    RECENT EXPENSES:\n"
    for e in expenses[:10]:
        context += f"    - {e['item']}: Rs. {e['amount']} ({e['category']})\n"
    
    return context
\end{lstlisting}

\subsection{Chat Input Handler}

\begin{lstlisting}[language=JavaScript, caption=Chat Message Handler (Chat.jsx)]
const handleSendMessage = async () => {
    if (!input.trim()) return;
    
    // Add user message to chat
    const userMessage = {
        id: Date.now(),
        type: 'user',
        content: input,
        timestamp: new Date()
    };
    setMessages(prev => [...prev, userMessage]);
    setInput('');
    setLoading(true);
    
    try {
        // Send to backend for parsing
        const response = await axios.post(
            `${API_BASE_URL}/api/expenses/parse`,
            {
                text: input,
                group_id: currentGroup?.id
            }
        );
        
        // Handle parsed expenses
        if (response.data.expenses?.length > 0) {
            for (const expense of response.data.expenses) {
                await saveExpense(expense);
            }
        }
        
        // Add bot response
        const botMessage = {
            id: Date.now() + 1,
            type: 'bot',
            content: response.data.reply,
            expenses: response.data.expenses,
            timestamp: new Date()
        };
        setMessages(prev => [...prev, botMessage]);
        
    } catch (error) {
        console.error('Parse error:', error);
        // Add error message
        setMessages(prev => [...prev, {
            id: Date.now() + 1,
            type: 'error',
            content: 'Sorry, I could not process that. Please try again.',
            timestamp: new Date()
        }]);
    } finally {
        setLoading(false);
    }
};
\end{lstlisting}
